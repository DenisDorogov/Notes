-------------------------<Частоупотребляемые команды Linux>------------------------------
ls [-l -la] //просмотр содержимого каталогов
cat - // печатает содержимое файла, переданного в параметре, в стандартный вывод.
cat -n //... определённое количество строк
cat > file.txt // перенаправляет вывод в файл
cd  //позволяет перейти из текущего каталога в указанный.
cd .. // возвращает на уровень вверх относительно текущего каталога.
cd - // возвращает к предыдущему каталогу.
pwd // печатает на экран текущий каталог.
mkdir  // создание новых каталогов. 
mkdir -p newdir/newdir1/newdir2  //создать всю структуру подкаталогов одной командой
file // показывает тип файла.
cp // копирование файлов и каталогов. (-r) Рекурсивно
cp -r file /home/file.txt 
mv file /home/file.txt // перемещение или переименование файлов и каталогов
rm // удаляет файлы и папки (-rf) Рекурсивно и без запроса
ln // создает жёсткие или символические ссылки на файлы. 
chmod // изменяет права доступа к файлу. Это чтение, запись и выполнение.(ugo+rwx)
chown // изменяет владельца файла. Только суперпользователь может изменять владельцев.
find // поиск в файловой системе, файлах и папках. 
locate // ведёт поиск в базе данных updatedb для шаблонов имён файлов.
du // показывает размер файла или каталога.
du -hsd // читаемый формат, компактно, глубина рекурсии
df // анализатор дискового пространства, (-h) читаемо.
dd // команда терминала для копирования и преобразования файлов. (с опциями)
mount / unmount // для подключения и отключения файловых систем Linux
less / more //  для просмотра длинных текстов, которые не вмещаются на одном экране.
cat longtext.txt | less
head / tail //Утилита выводит несколько первых/последних строк из файла.
grep // ищет текст по шаблону
ps | grep net
sort //сортировка вывода (-n) по числам, (-r) реверсивно
wc // для подсчёта количества слов, строк, байт и символов.
diff // Показывает различия между двумя файлами в построчном сравнении.
kill / xkill / pkill / killall // служат для завершения процессов
ps // печатает информацию о запущенных процессах. (aux) полная информация о процессах
pgrep //вы задаете имя процесса, а утилита показывает его идентификатор
htop //не только просматривать, но и контролировать процессы через его интерфейс 
time //время выполнения процесса. это секундомер для выполнения программы
sudo //выполняет команду от root имени
su // $ su опции пользователь
date //выводит дату и время в стандартный вывод
alias //команда создаёт синонимы для других команд linux
sleep // например, если вы хотите выключить компьютер через определенный промежуток 
		времени или использовать в качестве импровизированной тревоги
useradd / userdel / usermod //  добавлять, удалять и изменять уч. записи пользователей.
passwd // эта команда позволяет изменить пароль учетной записи пользователя.
man // команда man открывает руководство по определённой команде. 
whatis //показывает, какие разделы руководств есть для данной команды.
ip -a //для управления сетевыми подключениями
ping //отправляет эхозапросы на любой хост
nethogs // сколько трафика использует какая-либо программа в linux
traceroute // отслеживание маршрута
